function bingo(ticket, win) {
  let count = 0;
  let bin = new Map(ticket);
  for (let key of bin.keys()) {
    let num = String.fromCharCode(bin.get(key));
    if (key.includes(num)) {
      count++;
    }
  }
  if (win <= count) {
    return 'Winner!';
  } else {
    return 'Loser!';
  }
}

function toWeirdCase(string) {
  const arr = string.split('');
  let newArr = [];
  for (let i = 0; i < arr.length; i++) {
      if (arr[i] != ' ') {
      newArr.push(arr[i].toUpperCase());
      newArr.push(arr[i + 1]);
      i += 1;
    } else {
      newArr.push(' ');
    }
  }
  return newArr.join('');
}
